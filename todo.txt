void Game::HandleInput(){
    int keypressed = GetKeyPressed();
    switch (keypressed){
        case KEY_LEFT:
            MoveBlockLeft();
            break;
        case KEY_RIGHT:
            MoveBlockRight();
            break;
        case KEY_DOWN:
            MoveBlockDown();
            break;
        case KEY_X:
            RotateBlock();
            break;
        case KEY_Z:
            RotateBlockLeft();
            break;
        case KEY_C:
            Hold();
            break;
        case KEY_A:
            RotateBlock180();
            break;
        case KEY_SPACE:
            HardDrop();
            break;
    }
}
